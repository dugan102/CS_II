#include <iostream>
using namespace std;

class LLData {
private:
    int data;
    LLData *next;
    LLData *prev;
    LLData() {}
public:
    LLData(int d, LLData *nxt, LLData *prv) {
        data = d;
        next = nxt;
        prev = prv;
    }
    int getData() { return data; }
    LLData* getNext() { return next; }
    LLData* getPrev() { return prev; }
    void setData(int d) { data = d; }
    void setNext(LLData* nxt) { next = nxt; }
    void setPrev(LLData* prv) { prev = prv; }
};

class LinkedList {
private:
    LLData* first;
    LLData* last;
public:
    LinkedList() { first = NULL; last = NULL;}
    LLData *getFirst() { return first; }
    LLData *getLast() { return last; }
    void addItem(int d) {
        first = new LLData(d,first, nullptr);
        if (first->getNext() != nullptr){
            first->getNext()->setPrev(first);
        }

    }
    void showList() {
        LLData *curr = getFirst();
        while (curr != NULL) {
            cout << curr->getData() << " -> ";
            curr = curr->getNext();
        }
        cout << endl;
    }
    void showReverseList() {
        LLData *curr = getLast();
        while (curr != NULL) {
            cout << curr->getData() << " -> ";
            curr = curr->getPrev();
        }
        cout << endl;

    }
    void addItemAtEnd(int d) {
        if (first == NULL) {
            first = new LLData(d, nullptr, nullptr);
            //first->setNext(last);
            //last->setPrev(first);
            last = new LLData(d, nullptr, nullptr);
        }
        else {
            LLData* curr = first;
            while (curr->getNext() != NULL) {
                curr = curr->getNext();
            }
            curr->setNext(new LLData(d, nullptr,curr));
            last = new LLData(d, nullptr,curr);
        }
    }
    void deleteItem(int num){
        LLData* curr = first;
        while (curr != NULL){
            if (curr->getData() == num){
                if (curr->getPrev() == nullptr){
                    curr->getNext()->setPrev(nullptr);
                    first = curr->getNext();
                    delete curr;
                    break;
                }
                if (curr->getNext() == nullptr){
                    curr->getPrev()->setNext(nullptr);
                    last = curr->getPrev();
                    delete curr;
                    break;
                }
                if (curr->getPrev() != nullptr && curr->getNext() != nullptr){
                    curr->getNext()->setPrev(curr->getPrev());
                    curr->getPrev()->setNext(curr->getNext());
                    if (curr->getNext()->getNext() == nullptr){
                        last = curr->getNext();
                    }
                    delete curr;
                    break;
                }
            }
            curr = curr->getNext();
        }
    }
};

int main() {
    LinkedList myList;
    myList.addItemAtEnd(1);
    myList.addItemAtEnd(4);
    myList.addItem(17);
    myList.addItem(3);
    myList.addItemAtEnd(12);
    myList.addItemAtEnd(11);
    myList.showList();
    myList.showReverseList();
    myList.showList();
    myList.deleteItem(3);
    myList.showList();
    myList.addItem(11);
    myList.deleteItem(11);
    myList.deleteItem(12);
    myList.deleteItem(17);
    myList.deleteItem(4);

    myList.showList();
    myList.showReverseList();
    cout << myList.getFirst()->getData() << " " << myList.getLast()->getData();
}
